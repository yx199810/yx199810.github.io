  var buttonList = document.getElementsByTagName("li");/*得到按钮的节点数组*/
 var inner = document.getElementById("inner");/*得到图片的节点数组*/
 var perWidth = inner.children[0].offsetWidth;/*图片的高度和宽度*/
 function tab() {/*上一张图片*/
     inner.style.left = - perWidth * index + "px";/*根据index的值，实现图片转换*/
     for(var j = 0; j < buttonList.length; j++) {
         buttonList[j].className = "";
     }
     buttonList[index].className = "active";/*转换后按钮的颜色变换*/
 }
 for(var i = 0; i < buttonList.length; i++) {/*给每个按钮绑定事件*/
     buttonList[i].index = i;
     buttonList[i].onclick = function() {
         index = this.index;
         tab();/*调用tab函数，定位到对应下标的图片位置*/
     }
 }
 var index = 0;
 function next() {/*下一张图片*/
     index ++;
     if(index > buttonList.length - 1) {/*实现轮播*/
     index = 0;
     }
     tab();
 }
 var timer = setInterval(next,3000);/*实现图片转换的定时器*/

 inner.onmouseover = function() {
 clearInterval(timer);/*鼠标移入的时候，清理定时器*/
}
 inner.onmouseout = function() {
 timer = setInterval(next,3000);/*鼠标移出的时候，重置定时器*/
